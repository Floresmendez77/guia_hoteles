Variables & Mixins
sass/main.scss 
// ── Variables de marca ──────────────
$color-primario:   #1a3c5e;
$color-secundario: #c9a96e;
$ff-display: 'Playfair Display', serif;
$ff-ui:      'Montserrat', sans-serif;
$transition: .35s cubic-bezier(.4,0,.2,1);

// ── Mixin: btn-estilo ────────────────
@mixin btn-estilo(
  $bg:     transparent,
  $color:  $color-secundario
) {
  font-family: $ff-ui;
  font-size: .78rem;
  letter-spacing: .15em;
  text-transform: uppercase;
  background: $bg;
  color: $color;
  border: 1px solid $color-secundario;
  transition: $transition;
}

// ── Mixin: sombra-suave ──────────────
@mixin sombra-suave(
  $color: rgba(0,0,0,.12),
  $y:     8px,
  $blur:  40px
) {
  box-shadow: 0 #{$y} #{$blur} $color;
}
Nesting BEM & Media Queries
sass — nesting BEM 
// ── Nesting BEM .hotel-card ──────────
.hotel-card {
  background: #fff;
  @include sombra-suave();
  overflow: hidden;

  // BEM Elements
  &__img   { height: 220px; object-fit: cover; }
  &__title { font-family: $ff-display; }
  &__btn   { @include btn-estilo($color-primario, #fff); }

  // Estado hover
  &:hover {
    transform: translateY(-6px);
    @include sombra-suave(rgba($color-primario, .18));
  }
}

// ── Media Queries ────────────────────
@media (max-width: 768px) {
  #hoteles-container {
    grid-template-columns: 1fr;
  }
}

Operaciones Matemáticas
less — operaciones 
// ── Sistema de columnas ──────────────
@grid-columns:  12;
@grid-gutter:   30px;
@container-max: 1200px;

// Cálculo automático de columna
@col-width: (@container-max
  - (@grid-gutter * (@grid-columns - 1)))
  / @grid-columns;
// = (1200 - 330) / 12 = 72.5px

// Separador calculado
@sep-w: 60px;
.gold-divider {
  width: @sep-w;          // 60px
  &--grande { width: @sep-w * 2; } // 120px
}
